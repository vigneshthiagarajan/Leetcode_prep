WITH TEMP AS (SELECT E1.ID AS EMPLOYEEID, E1.NAME AS EMPLOYEE_NAME, E1.SALARY AS EMPLOYEE_SALARY, E2.NAME AS MANAGER_NAME, E2.SALARY AS MANAGER_SALARY FROM EMPLOYEE E1 LEFT OUTER JOIN EMPLOYEE E2 on E1.MANAGERID = E2.ID)
SELECT EMPLOYEE_NAME AS EMPLOYEE FROM TEMP WHERE EMPLOYEE_SALARY>MANAGER_SALARY;
