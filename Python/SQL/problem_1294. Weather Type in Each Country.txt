WITH TEMP AS (SELECT COUNTRY_ID, AVG(WEATHER_STATE) AS AVG_WEATHER FROM WEATHER WHERE DAY LIKE "2019-11%" GROUP BY COUNTRY_ID)
SELECT COUNTRY_NAME,
CASE
WHEN AVG_WEATHER <= 15  THEN "Cold"
WHEN AVG_WEATHER >=25 THEN "Hot"
ELSE "Warm"
END AS WEATHER_TYPE FROM COUNTRIES NATURAL JOIN TEMP;
